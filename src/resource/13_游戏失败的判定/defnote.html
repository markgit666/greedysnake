1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  5. import java.awt.event.ActionEvent;
  6. import java.awt.event.ActionListener;
  7. import java.awt.event.KeyAdapter;
  8. import java.awt.event.KeyEvent;
  9. import java.util.Random;
  11. /**
  12.  * @Auther: msb-zhaoss
  13.  */
  14. public class GamePanel extends JPanel {
  15.     //小蛇的长度：
  16.     int length;
  17.     //定义一个数组，专门存储小蛇的头和身体对应的x轴坐标
  18.     int[] snakeX = new int[500];//500小蛇长度
  19.     //定义一个数组，专门存储小蛇的头和身体对应的y轴坐标
  20.     int[] snakeY = new int[500];//500小蛇长度
  22.     //定义一个变量来判断小蛇的方向：
  23.     String direction = "R";//默认情况下小蛇是向右的
  24.     //上U下D左L右R
  26.     //定义一个变量判断游戏的状态：是开始的还是结束：
  27.     boolean isStart = false;//默认情况下游戏没有开始
  29.     //加入一个定时器：
  30.     Timer timer;
  32.     //定义食物的坐标：
  33.     int foodX;
  34.     int foodY;
  36.     //定义一个积分：
  37.     int score;
  39.     //定义一个变量 判断小蛇是否死亡：
  40.     boolean isDie = false;//默认情况下小蛇是活着
  42.     //初始化信息：
  43.     public void init(){
  44.         length = 3;
  45.         //初始化蛇头的信息：
  46.         snakeX[0] = 200;
  47.         snakeY[0] = 375;
  48.         //第一节身子的信息：
  49.         snakeX[1] = 175;
  50.         snakeY[1] = 375;
  51.         //第二节身子的信息：
  52.         snakeX[2] = 150;
  53.         snakeY[2] = 375;
  55.         //初始化食物信息：
  56.         foodX = 325;
  57.         foodY = 125;
  59.         //设置定时器，让定时器每100ms执行一次操作：
  60.         timer = new Timer(100, new ActionListener() {
  61.             //相当于每100ms执行一下actionPerformed中的动作：
  62.             @Override
  63.             public void actionPerformed(ActionEvent e) {
  64.                 //让小蛇动一下：
  65.                 //如果游戏是开始的情况下，小蛇才会动：
  66.                 //再加入一个判断：在游戏开始的状态下，小蛇是活着的
  67.                 if(isStart==true&&isDie==false){
  68.                     //注意：后一节跟着前一节动：
  69.                     //先动身子
  70.                     for(int i=length-1;i>0;i--){
  71.                         snakeX[i] = snakeX[i-1];
  72.                         snakeY[i] = snakeY[i-1];
  73.                     }
  74.                     //最后动头：
  75.                     switch (direction){
  76.                         case "R" :snakeX[0] = snakeX[0]+25;break;
  77.                         case "L" :snakeX[0] = snakeX[0]-25;break;
  78.                         case "U" :snakeY[0] = snakeY[0]-25;break;
  79.                         case "D" :snakeY[0] = snakeY[0]+25;break;
  80.                     }
  81.                     //防止小蛇出界：(上下左右都要进行限制)
  82.                     if(snakeX[0]>750){
  83.                         snakeX[0] = 25;
  84.                     }
  85.                     if(snakeX[0]<25){
  86.                         snakeX[0] = 750;
  87.                     }
  89.                     if(snakeY[0]<100){
  90.                         snakeY[0] = 725;
  91.                     }
  92.                     if(snakeY[0]>725){
  93.                         snakeY[0] = 100;
  94.                     }
  96.                     //当蛇头的坐标和食物的坐标一样的时候，就相当于碰撞了
  97.                     if(snakeX[0] == foodX && snakeY[0] == foodY){
  98.                         //蛇长度加一：
  99.                         length++;
  100.                         //食物的坐标换位置：
  101.                         foodX = 25 + 25*(new Random().nextInt(28));//25-725
  102.                         foodY = 100 + 25*(new Random().nextInt(25));;    //100-725
  104.                         //每碰撞一次 积分要加10：
  105.                         score = (length-3)*10;
  106.                     }
  108.                     //小蛇死亡判定：
  109.                     for(int i=1;i<length;i++){
  110.                         if(snakeX[0]==snakeX[i]&&snakeY[0]==snakeY[i]){
  111.                             isDie = true;
  112.                         }
  113.                     }
  114.                     repaint();
  115.                 }
  116.             }
  117.         });
  119.         //启动定时器
  120.         timer.start();
  121.     }
  123.     public GamePanel() {
  124.         init();
  125.         //让焦点在面板上：
  126.         this.setFocusable(true);//获取焦点
  127.         //给面板添加一个监听：
  128.         this.addKeyListener(new KeyAdapter() {
  129.             @Override
  130.             public void keyPressed(KeyEvent e) {
  131.                 super.keyPressed(e);
  132.                 int keyCode = e.getKeyCode();
  133.                 System.out.println(keyCode);
  134.                 if(keyCode==KeyEvent.VK_SPACE){//证明你按下的是空格键：
  135.                     if(isDie){//小蛇死了：
  136.                         //之前的数据全部恢复到初始化状态：
  137.                         init();
  138.                         isDie = false;
  139.                     }else{//小蛇没死的情况下
  140.                         isStart = !isStart;
  141.                         repaint();//重新绘制界面-->底层会去调用：paintComponent方法
  142.                     }
  143.                 }
  144.                 if(keyCode==KeyEvent.VK_UP){
  145.                     direction = "U";
  146.                 }
  147.                 if(keyCode==KeyEvent.VK_DOWN){
  148.                     direction = "D";
  149.                 }
  150.                 if(keyCode==KeyEvent.VK_LEFT){
  151.                     direction = "L";
  152.                 }
  153.                 if(keyCode==KeyEvent.VK_RIGHT){
  154.                     direction = "R";
  155.                 }
  158.             }
  159.         });
  160.     }
  162.     //paintComponent:方法不需要我们自己去调用，系统线程会去调用，
  163.     //这个方法作用：所有画的动作都在这个方法中执行：
  164.     @Override
  165.     protected void paintComponent(Graphics g) {
  166.         super.paintComponent(g);
  167.         //给面板设置一个背景色：
  168.         this.setBackground(new Color(235, 194, 168));
  169.         //画头部图片：
  170.         Images.headerImg.paintIcon(this,g,10,10);
  172.         //画一个矩形：
  173.         //换一下画笔的颜色：
  174.         g.setColor(new Color(213, 238, 232));
  175.         g.fillRect(10,70,770,685);
  177.         //根据小蛇的方向来画蛇头：
  178.         switch (direction){
  179.             case "L" : Images.leftImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  180.             case "R" : Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  181.             case "U" : Images.upImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  182.             case "D" : Images.downImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  183.         }
  185.         /*//画第一节身子：
  186.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  187.         //画第二节身子：
  188.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  189.         //对蛇身子的画法进行优化，加入循环：
  190.         for(int i=1;i<length;i++){
  191.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  192.         }
  194.         //画食物：
  195.         Images.foodImg.paintIcon(this,g,foodX,foodY);
  199.         //当游戏暂停的时候，在面板中画文字：
  200.         if(isStart==false){//游戏是暂停的时候，才绘制文字：
  201.             g.setColor(new Color(255, 186, 90));
  202.             //设置：字体，加粗，字号
  203.             g.setFont(new Font("微软雅黑",Font.BOLD,40));
  204.             g.drawString("点击空格游戏开始",250,330);
  205.         }
  209.         //画积分：
  210.         g.setColor(new Color(255, 255, 255));
  211.         //设置：字体，加粗，字号
  212.         g.setFont(new Font("微软雅黑",Font.BOLD,20));
  213.         g.drawString("积分:"+score,620,40);
  215.         //如果小蛇死亡，在页面中画入提示语：
  216.         if(isDie){
  217.             g.setColor(new Color(255, 56, 49));
  218.             //设置：字体，加粗，字号
  219.             g.setFont(new Font("微软雅黑",Font.BOLD,30));
  220.             g.drawString("小蛇撞到自己死亡了,游戏停止,按下空格重新开始游戏",40,330);
  221.         }
  223.     }
  224. }
