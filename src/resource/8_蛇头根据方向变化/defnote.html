1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  6. /**

  9. public class GamePanel extends JPanel {
  10.     //小蛇的长度：
  11.     int length;
  12.     //定义一个数组，专门存储小蛇的头和身体对应的x轴坐标
  13.     int[] snakeX = new int[500];//500小蛇长度
  14.     //定义一个数组，专门存储小蛇的头和身体对应的y轴坐标
  15.     int[] snakeY = new int[500];//500小蛇长度
  17.     //定义一个变量来判断小蛇的方向：
  18.     String direction = "R";//默认情况下小蛇是向右的
  19.     //上U下D左L右R
  21.     //初始化信息：
  22.     public void init(){
  23.         length = 3;
  24.         //初始化蛇头的信息：
  25.         snakeX[0] = 200;
  26.         snakeY[0] = 375;
  27.         //第一节身子的信息：
  28.         snakeX[1] = 175;
  29.         snakeY[1] = 375;
  30.         //第二节身子的信息：
  31.         snakeX[2] = 150;
  32.         snakeY[2] = 375;
  35.     }
  37.     public GamePanel() {
  38.         init();
  39.     }
  41.     //paintComponent:方法不需要我们自己去调用，系统线程会去调用，
  42.     //这个方法作用：所有画的动作都在这个方法中执行：
  43.     @Override
  44.     protected void paintComponent(Graphics g) {
  45.         super.paintComponent(g);
  46.         //给面板设置一个背景色：
  47.         this.setBackground(new Color(235, 194, 168));
  48.         //画头部图片：
  49.         Images.headerImg.paintIcon(this,g,10,10);
  51.         //画一个矩形：
  52.         //换一下画笔的颜色：
  53.         g.setColor(new Color(213, 238, 232));
  54.         g.fillRect(10,70,770,685);
  56.         //根据小蛇的方向来画蛇头：
  57.         switch (direction){
  58.             case "L" : Images.leftImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  59.             case "R" : Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  60.             case "U" : Images.upImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  61.             case "D" : Images.downImg.paintIcon(this,g,snakeX[0],snakeY[0]);break;
  62.         }
  64.         /*//画第一节身子：
  65.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  66.         //画第二节身子：
  67.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  68.         //对蛇身子的画法进行优化，加入循环：
  69.         for(int i=1;i<length;i++){
  70.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  71.         }
  72.     }
  73. }
